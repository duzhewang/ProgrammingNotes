class Solution:
    def highFive(self, items: List[List[int]]) -> List[List[int]]:
        dict={}
        for item in items:
            if item[0] not in dict:
                dict[item[0]]=[item[1]]
            else:
                dict[item[0]].append(item[1])
        res=[]
        for k, v in dict.items():
            res.append([k, sum(list(reversed(sorted(v)))[:5])//5])
        return res
