class Solution:
    def findNumbers(self, nums: List[int]) -> int:
        digits=[]
        for num in nums:
            d=0
            while num>0:
                num=num//10
                d+=1
            digits.append(d)
            
        count=0
        for d in digits:
            if d%2==0:
                count+=1
            else:
                count
        return count
