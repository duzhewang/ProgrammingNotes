class ParkingSystem:

    def __init__(self, big: int, medium: int, small: int):
        self.dic={1:big, 2:medium, 3:small}
        

    def addCar(self, carType: int) -> bool:
        if self.dic[carType]>=1:
           self.dic[carType]-=1
           return True
        else:
            return False
