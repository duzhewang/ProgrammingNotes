Reverse strings:

class Solution:
    def reverseString(self, s: List[str]) -> None:
        """
        Do not return anything, modify s in-place instead.
        """
        n=len(s)
        if n%2==0:
           for a in range(n//2):
               s[a], s[n-1-a]=s[n-1-a], s[a]
        else:
           for a in range((n-1)//2):
                s[a], s[n-1-a]=s[n-1-a], s[a]
